# minutes:  calculate, for a given phone number, the month with the maximum number of minutes
#
#!/bin/csh
#
#
set max=0
set month = " "

cd d_$1
foreach mnth (*)
	set minutes = `cat $mnth |../calc 2`
	   if ( $max < $minutes) then
		set month = $mnth
		set max = $minutes
	   endif
end

echo $month $max
